<?php

namespace App\Services\Subscription;

use App\Models\SubscriptionRenewal;
use App\Models\User;
use App\Models\Plan;
use App\Models\Subscription;
use Carbon\Carbon;
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Str;

class OldSubscriptionService
{
  public function subscribe(User $user, Plan $plan, string $billingPeriod = 'monthly'): Subscription
  {
    return DB::transaction(function () use ($user, $plan, $billingPeriod) {
      $currentSubscription = $user->activeSubscription();
      $pendingSubscription = $user->pendingSubscription;

      // If there's a pending subscription, we can't create a new one
      if ($pendingSubscription) {
        throw new \Exception('You already have a pending subscription change.');
      }

      // Calculate subscription details
      $startDate = now();
      $price = $this->calculatePrice($plan, $billingPeriod);

      // If user has an active subscription
      if ($currentSubscription) {
        // If switching to free plan, cancel current subscription
        if ($plan->price === 0) {
          return $this->cancel($user);
        }

        // Determine if this is an upgrade or downgrade
        $isUpgrade = $plan->price > $currentSubscription->plan->price;

        // For upgrades, take effect immediately
        // For downgrades, take effect after current subscription ends
        $startDate = $isUpgrade
          ? now()
          : $currentSubscription->ends_at;

        // If it's a downgrade, create pending subscription
        if (!$isUpgrade) {
          return $this->createSubscription($user, $plan, [
            'status' => Subscription::STATUS_PENDING,
            'starts_at' => $startDate,
            'billing_period' => $billingPeriod,
            'price' => $price
          ]);
        }

        // For upgrades, cancel current subscription first
        $currentSubscription->update([
          'ends_at' => now(),
          'status' => Subscription::STATUS_CANCELLED
        ]);
      }

      // Create new active subscription
      return $this->createSubscription($user, $plan, [
        'status' => Subscription::STATUS_ACTIVE,
        'starts_at' => $startDate,
        'billing_period' => $billingPeriod,
        'price' => $price
      ]);
    });
  }

  public function cancel(User $user): bool
  {
    return DB::transaction(function () use ($user) {
      $subscription = $user->activeSubscription();

      if (!$subscription) {
        return false;
      }

      // Cancel any pending subscriptions
      $user->pendingSubscription?->update([
        'status' => Subscription::STATUS_CANCELLED,
        'ends_at' => now()
      ]);

      // Cancel active subscription
      $subscription->update([
        'ends_at' => $subscription->billing_period === 'yearly'
          ? $subscription->starts_at->addYear()
          : $subscription->starts_at->addMonth(),
        'status' => Subscription::STATUS_CANCELLED
      ]);

      // Revert to free plan after current subscription ends
      $freePlan = Plan::where('price', 0)->first();
      if ($freePlan) {
        $this->createSubscription($user, $freePlan, [
          'status' => Subscription::STATUS_PENDING,
          'starts_at' => $subscription->ends_at,
          'billing_period' => 'monthly',
          'price' => 0
        ]);
      }

      return true;
    });
  }

  private function createSubscription(User $user, Plan $plan, array $attributes): Subscription
  {
    $endDate = $attributes['billing_period'] === 'yearly'
      ? Carbon::parse($attributes['starts_at'])->addYear()
      : Carbon::parse($attributes['starts_at'])->addMonth();

    $subscription = $user->subscription()->create([
      'plan_id' => $plan->id,
      'status' => Subscription::STATUS_PENDING,
      'starts_at' => $attributes['starts_at'],
      'ends_at' => $endDate,
      'billing_period' => $attributes['billing_period'],
      'price' => $attributes['price'],
      'currency' => $plan->currency
    ]);

    /*if ($attributes['status'] === Subscription::STATUS_ACTIVE) {
      // Create renewal record for active subscriptions
      $subscription->renewals()->create([
        'amount' => $attributes['price'],
        'currency' => $plan->currency,
        'status' => 'successful',
        'paychangu_reference' => Str::uuid(),
        'billing_period' => $attributes['billing_period']
      ]);

      // Update user settings
      $user->settings()->updateOrCreate(
        ['user_id' => $user->id],
        ['plan_id' => $plan->id]
      );
    }*/

    // Create a pending renewal record
    $subscription->renewals()->create([
      'amount' => $attributes['price'],
      'currency' => $plan->currency,
      'status' => SubscriptionRenewal::STATUS_PENDING,
      'billing_period' => $attributes['billing_period'],
      // paychangu_reference will be updated when payment is processed
    ]);

    return $subscription->load('renewals', 'plan');
  }

  private function calculatePrice(Plan $plan, string $billingPeriod): float
  {
    if ($billingPeriod === 'yearly') {
      return $plan->price * 12 * 0.8; // 20% discount for yearly
    }
    return $plan->price;
  }

  /**
   * Handle successful payment from Paychangu
   */
  public function handlePaymentSuccess(string $paychanguReference, float $amount): Subscription
  {
    return DB::transaction(function () use ($paychanguReference, $amount) {
      $renewal = SubscriptionRenewal::where('status', SubscriptionRenewal::STATUS_PENDING)
        ->whereNull('paychangu_reference')
        ->where('amount', $amount)
        ->latest()
        ->first();

      if (!$renewal) {
        throw new \Exception('No matching pending renewal found');
      }

      // Update the renewal with the paychangu reference
      $renewal->update([
        'paychangu_reference' => $paychanguReference,
        'status' => SubscriptionRenewal::STATUS_SUCCESSFUL
      ]);

      $subscription = $renewal->subscription;

      // Activate the subscription
      $subscription->update([
        'status' => Subscription::STATUS_ACTIVE
      ]);

      // Update user settings
      $subscription->user->settings()->updateOrCreate(
        ['user_id' => $subscription->user_id],
        ['plan_id' => $subscription->plan_id]
      );

      return $subscription->fresh()->load('renewals', 'plan');
    });
  }

  /**
   * Handle failed payment from Paychangu
   */
  public function handlePaymentFailure(string $paychanguReference, ?string $failureReason = null): void
  {
    $renewal = SubscriptionRenewal::where('status', SubscriptionRenewal::STATUS_PENDING)
      ->whereNull('paychangu_reference')
      ->latest()
      ->first();

    if ($renewal) {
      DB::transaction(function () use ($renewal, $paychanguReference, $failureReason) {
        $renewal->update([
          'paychangu_reference' => $paychanguReference,
          'status' => SubscriptionRenewal::STATUS_FAILED,
          'failure_reason' => $failureReason
        ]);

        // Cancel the pending subscription
        $renewal->subscription->update([
          'status' => Subscription::STATUS_CANCELLED,
          'ends_at' => now()
        ]);
      });
    }
  }
}
